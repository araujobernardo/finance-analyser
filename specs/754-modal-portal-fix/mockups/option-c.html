Component tree (simplified)
App.tsx
 ├─ Sidebar ← triggers modal via useModal()
 ├─ Router / Pages
 │   └─ DuplicateWarningModal ← trigger via useModal()

 └─ ModalHost ← single portal mount on document.body
     └─ createPortal(activeModal, document.body)
// useModal.ts — shared context hook
const useModal = () => useContext(ModalContext);

// In Sidebar.tsx — no more rendering modal inline
const { open } = useModal();
<button onClick={() => open('add-account')}>+ Add Account</button>

// ModalHost.tsx — single portal at App root
function ModalHost() {
  const { activeModal } = useModal();
  return createPortal(<ActiveModal />, document.body);
}
